        const DATA_PATH = window.location.hostname.includes('github.io') ? './data/' : './';

        async function loadTips() {
            const container = document.getElementById('tips-container');
            
            try {
                let response = await fetch(DATA_PATH + 'tips.json');
                if (!response.ok) {
                    response = await fetch('./tips.json');
                }
                const tips = await response.json();
                
                if (!tips || tips.length === 0) {
                    showEmptyState(container, 'No tips available yet. Check back soon!');
                    return;
                }

                renderTips(tips);
                updateStats(tips);
                setupFilters(tips);
                
            } catch (error) {
                console.error('Failed to load tips:', error);
                showEmptyState(container, 'Unable to load tips. Please refresh the page.');
            }
        }

        async function loadBuilds() {
            const container = document.getElementById('builds-container');
            
            try {
                let response = await fetch(DATA_PATH + 'builds.json');
                if (!response.ok) {
                    response = await fetch('./builds.json');
                }
                const builds = await response.json();
                renderBuilds(builds);
            } catch (error) {
                console.error('Failed to load builds:', error);
            }
        }

        function renderTips(tips) {
            const container = document.getElementById('tips-container');
            
            container.innerHTML = tips.map(tip => {
                const cardClasses = ['tip-card'];
                if (tip.hot) cardClasses.push('hot');
                if (tip.tags.includes('meta')) cardClasses.push('meta');
                
                return `
                    <article class="${cardClasses.join(' ')}" data-tags="${tip.tags.join(' ')}" data-hot="${tip.hot}">
                        <div class="tip-tags">
                            ${tip.isNew ? '<span class="tag new">NEW</span>' : ''}
                            ${tip.hot ? '<span class="tag hot">üî• HOT</span>' : ''}
                            ${tip.tags.map(tag => `<span class="tag ${tag}">${tag}</span>`).join('')}
                        </div>
                        <h3 class="tip-title">${escapeHtml(tip.title)}</h3>
                        <p class="tip-content">${escapeHtml(tip.content)}</p>
                        <div class="tip-footer">
                            <div class="tip-source">
                                ${tip.source}
                                ${tip.url ? ` ‚Ä¢ <a href="${tip.url}" target="_blank" rel="noopener">View Source</a>` : ''}
                            </div>
                            <div class="tip-score">‚ñ≤ ${tip.score || 0}</div>
                        </div>
                    </article>
                `;
            }).join('');
        }

        function updateStats(tips) {
            document.getElementById('total-tips').textContent = tips.length;
            document.getElementById('hot-tips').textContent = tips.filter(t => t.hot).length;
            document.getElementById('tips-count').textContent = `${tips.length} tips`;
            
            const lastUpdate = tips[0]?.scraped_at;
            if (lastUpdate) {
                const date = new Date(lastUpdate);
                document.getElementById('last-update').textContent = date.toLocaleDateString('en-US', {
                    month: 'short',
                    day: 'numeric',
                    year: 'numeric'
                });
            }
        }

        function setupFilters(tips) {
            document.querySelectorAll('.quick-link').forEach(btn => {
                btn.addEventListener('click', () => {
                    document.querySelectorAll('.quick-link').forEach(b => b.classList.remove('active'));
                    btn.classList.add('active');
                    
                    const filter = btn.dataset.filter;
                    filterTips(filter);
                });
            });
        }

        function filterTips(filter) {
            const cards = document.querySelectorAll('.tip-card');
            let visible = 0;

            cards.forEach(card => {
                const tags = card.dataset.tags || '';
                const isHot = card.dataset.hot === 'true';
                
                let show = filter === 'all';
                if (filter === 'hot' && isHot) show = true;
                if (filter !== 'all' && filter !== 'hot' && tags.includes(filter)) show = true;

                card.style.display = show ? '' : 'none';
                if (show) visible++;
            });

            document.getElementById('tips-count').textContent = `${visible} tips`;
        }

                    <p>${message}</p>
                </div>
            `;
        }

        function escapeHtml(text) {
            if (!text) return '';
            const div = document.createElement('div');
            div.textContent = text;
            return div.innerHTML;
        }
